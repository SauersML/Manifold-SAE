# Find the torch-bundled cuBLAS and preload it so glibc dlopen uses it
# for every subsequent cuBLAS load (system + bundled both map to one
# file handle). Resolves gamfit's dual-stack conflict.
import nvidia.cublas, os
CUBLAS = os.path.join(os.path.dirname(nvidia.cublas.__file__), "lib", "libcublas.so.12")
print("LD_PRELOAD:", CUBLAS)
